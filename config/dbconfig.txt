## Конфигурация соединения с MySQL
## Это используется для статистики, сбора отзывов,
## администрирования и игровой библиотеки.

## Должен ли быть включен SQL? Снимите комментарий, чтобы включить.
#SQL_ENABLED

## Server the MySQL database can be found at.
## Examples: localhost, 200.135.5.43, www.mysqldb.com, etc.
ADDRESS localhost

## Порт сервера MySQL (по умолчанию 3306).
PORT 3306

## База данных для всех SQL-функций, а не только для обратной связи.
FEEDBACK_DATABASE feedback

## Префикс, который будет добавлен к имени каждой таблицы, старые базы данных требуют, чтобы он был установлен в erro_
## Обратите внимание, это не изменяет имена таблиц в базе данных, вам придется сделать это самостоятельно.
##IE:
## FEEDBACK_TABLEPREFIX
## FEEDBACK_TABLEPREFIX SS13_
## Удалите «SS13_», если вы используете стандартный файл схемы.
FEEDBACK_TABLEPREFIX SS13_

## Имя пользователя/логин, используемый для доступа к базе данных.
FEEDBACK_LOGIN username

## Пароль, используемый для доступа к базе данных.
FEEDBACK_PASSWORD password

## Время в секундах для таймаута асинхронных запросов.
## Установите значение 0 для бесконечного выполнения
ASYNC_QUERY_TIMEOUT 10

## Время в секундах для выполнения блокирующих запросов до медленного таймаута запроса
## Установите значение 0 для бесконечности
## Должно быть меньше или равно ASYNC_QUERY_TIMEOUT
BLOCKING_QUERY_TIMEOUT 5

## Минимальное количество sql-соединений, которое необходимо поддерживать в пуле. Установка этого значения на серверах, географически удаленных от базы данных, может повысить производительность.
POOLING_MIN_SQL_CONNECTIONS 1

## Максимальное количество sql-соединений с базой данных.
POOLING_MAX_SQL_CONNECTIONS 25

## Максимальное количество одновременных асинхронных запросов, которые могут ожидать результата, прежде чем мы начнем ставить в очередь дальнейшие запросы к базе данных.
MAX_CONCURRENT_QUERIES 25
